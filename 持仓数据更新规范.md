## 🔧 脚本规范

### 所有脚本必须包含的读取逻辑

```python
import json
import os

def load_latest_holdings():
    """加载最新持仓数据"""
    holdings_file = "/home/admin/openclaw/workspace/holdings_current.json"
    
    if not os.path.exists(holdings_file):
        raise FileNotFoundError("❌ 持仓文件不存在！请老板先发送持仓截图进行 OCR 识别。")
    
    with open(holdings_file, 'r', encoding='utf-8') as f:
        data = json.load(f)
    
    # 检查数据是否过期（超过 7 天）
    from datetime import datetime, timedelta
    last_updated = datetime.fromisoformat(data['last_updated'])
    if datetime.now().astimezone() - last_updated > timedelta(days=7):
        print(f"⚠️ 警告：持仓数据已过期 {last_updated}，建议更新")
    
    return data

# 使用示例
holdings = load_latest_holdings()
print(f"总市值：{holdings['total_market_value']}")
```
