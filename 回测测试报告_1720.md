**原因**:

```python
# 从 akshare 获取真实历史价格
import akshare as ak

def get_historical_prices(etf_codes, start_date, end_date):
    """获取真实历史价格"""
    prices = {}
    for code in etf_codes:
        df = ak.fund_etf_hist_em(symbol=code, period="daily", 
                                  start_date=start_date, end_date=end_date)
        prices[code] = df.set_index('日期')['收盘'].to_dict()
    return prices
```

### 方案 2: 修复回测引擎逻辑

```python
# 修复资金计算
def execute_rebalance(self, date, target_etfs):
    # 卖出：按真实价格计算
    for code in list(self.positions.keys()):
        if code not in target_etfs:
            shares = self.positions[code]
            price = self.historical_data[code][date]['close']
            self.capital += shares * price * (1 - TRANSACTION_COST)
            del self.positions[code]
    
    # 买入：按真实价格计算
    for code in target_etfs:
        price = self.historical_data[code][date]['close']
        target_value = self.capital / len(target_etfs)
        shares = int(target_value / price)
        self.positions[code] = shares
        self.capital -= shares * price * (1 + TRANSACTION_COST)
```
